% load and configure fonts (font setup inspired by https://github.com/posquit0/)
\def\@fontdir{fonts/} % define '/fonts' directory
\newfontfamily\FA[Path=\@fontdir]{FontAwesome} % use ttf fontawesome font
\newfontfamily\headerfont[ % use Roboto font for header
  Path=\@fontdir,
  UprightFont=*-Regular,
  ItalicFont=*-Italic,
  BoldFont=*-Bold,
  BoldItalicFont=*-BoldItalic,
]{Roboto}
\newfontfamily\headerfontlight[
  Path=\@fontdir,
  UprightFont=*-Thin,
  ItalicFont=*-ThinItalic,
  BoldFont=*-Medium,
  BoldItalicFont=*-MediumItalic,
]{Roboto}
