% draw a horizontal line with specific thickness
\newcommand*{\customsection}[1]{
  \noindent
  \adjustbox{margin=0 .15cm 0 .25cm}{
    \sectionstyle{
      \begin{tikzpicture}[]
        \fill[MaterialLightBlue300, anchor=north west] (0, -1.6ex) rectangle (\linewidth, -1.6ex + -0.28cm);
        \node[anchor=north west] (txt) at (0, 0) {#1};
      \end{tikzpicture}
    }
    \phantomsection
    \newline
  }
}
